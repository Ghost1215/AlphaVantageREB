def quote file: nil, datatype: @datatype
  check_datatype(datatype, file)
  url = "function=GLOBAL_QUOTE&symbol=#{symbol}"
  return @client.download(url, file) if datatype == "csv"
  return open_struct(url, "Global Quote")
end
